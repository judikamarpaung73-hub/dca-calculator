<script>
    function hitungAlokasi() {
        // Ambil nilai dari input box
        let modalInput = document.getElementById("modal").value;
        let modal = parseFloat(modalInput);

        // Validasi jika input kosong atau negatif
        if (isNaN(modal) || modal < 0) {
            modal = 0;
        }

        // Variabel porsi porto
        const porsi = {
            vti: 0.55,
            vgt: 0.15,
            bbca: 0.15,
            vxus: 0.10,
            btc: 0.05
        };

        // Hitung nominal masing-masing aset
        let danaVTI = modal * porsi.vti;
        let danaVGT = modal * porsi.vgt;
        let danaBBCA = modal * porsi.bbca;
        let danaVXUS = modal * porsi.vxus;
        let danaBTC = modal * porsi.btc;

        // Format angka ke format mata uang Rupiah
        const formatRupiah = new Intl.NumberFormat('id-ID', {
            style: 'currency',
            currency: 'IDR',
            minimumFractionDigits: 0,
            maximumFractionDigits: 0
        });

        // Tampilkan hasil hitungan ke dalam tabel HTML
        document.getElementById("val-vti").innerText = formatRupiah.format(danaVTI);
        document.getElementById("val-vgt").innerText = formatRupiah.format(danaVGT);
        document.getElementById("val-bbca").innerText = formatRupiah.format(danaBBCA);
        document.getElementById("val-vxus").innerText = formatRupiah.format(danaVXUS);
        document.getElementById("val-btc").innerText = formatRupiah.format(danaBTC);
        document.getElementById("val-total").innerText = formatRupiah.format(modal);
    }
</script>
